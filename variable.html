var answer = window.prompt("ヘルプを見ますか？？？");
if (answer === "はい") {
	window.alert("自分でGoogle検索して調べなさい！");
} else if (answer === "いいえ") {
	window.alert("ゲーム起動中...");
} else {
	window.alert("'はい'か'いいえ'でお答えください。");
}
